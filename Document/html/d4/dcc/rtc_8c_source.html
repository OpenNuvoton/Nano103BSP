123int32_t RTC_Open(S_RTC_TIME_DATA_T *sPt)
124{
125 uint32_t u32TimeOutCount = SystemCoreClock; // 1 second timeout
126 uint32_t u32Reg;
127
128 RTC->INIT = RTC_INIT_KEY;
129
130 if(RTC->INIT != 0x1)
131 {
132 RTC->INIT = RTC_INIT_KEY;
133
134 while(RTC->INIT != 0x1)
135 {
136 if(u32TimeOutCount == 0) return -1;
137 u32TimeOutCount--;
138 }
139 }
140
141 if(sPt == NULL)
142 return -1;
143
144 /*-----------------------------------------------------------------------------------------------------*/
145 /* Second, set RTC 24/12 hour setting */
146 /*-----------------------------------------------------------------------------------------------------*/
147 if (sPt->u32TimeScale == RTC_CLOCK_12)
148 {
149 RTC_RWEN();
150 RTC->CLKFMT &= ~RTC_CLKFMT_24HEN_Msk;
151
152 /*-------------------------------------------------------------------------------------------------*/
153 /* important, range of 12-hour PM mode is 21 up to 32 */
154 /*-------------------------------------------------------------------------------------------------*/
155 if (sPt->u32AmPm == RTC_PM)
156 sPt->u32Hour += 20;
157 }
158 else
159 {
160 RTC_RWEN();
161 RTC->CLKFMT |= RTC_CLKFMT_24HEN_Msk;
162 }
163
164 /*-----------------------------------------------------------------------------------------------------*/
165 /* Set RTC Calender Loading */
166 /*-----------------------------------------------------------------------------------------------------*/
167 u32Reg = ((sPt->u32Year - RTC_YEAR2000) / 10) << 20;
168 u32Reg |= (((sPt->u32Year - RTC_YEAR2000) % 10) << 16);
169 u32Reg |= ((sPt->u32Month / 10) << 12);
170 u32Reg |= ((sPt->u32Month % 10) << 8);
171 u32Reg |= ((sPt->u32Day / 10) << 4);
172 u32Reg |= (sPt->u32Day % 10);
173 g_u32Reg = u32Reg;
174
175 RTC_RWEN();
176 RTC->CAL = (uint32_t)g_u32Reg;
177
178 /*-----------------------------------------------------------------------------------------------------*/
179 /* Set RTC Time Loading */
180 /*-----------------------------------------------------------------------------------------------------*/
181 u32Reg = ((sPt->u32Hour / 10) << 20);
182 u32Reg |= ((sPt->u32Hour % 10) << 16);
183 u32Reg |= ((sPt->u32Minute / 10) << 12);
184 u32Reg |= ((sPt->u32Minute % 10) << 8);
185 u32Reg |= ((sPt->u32Second / 10) << 4);
186 u32Reg |= (sPt->u32Second % 10);
187 g_u32Reg = u32Reg;
188
189 RTC_RWEN();
190 RTC->TIME = (uint32_t)g_u32Reg;
191
192 RTC_RWEN();
193 RTC->WEEKDAY = sPt->u32DayOfWeek;
194
195 return 0;

216void RTC_GetDateAndTime(S_RTC_TIME_DATA_T *sPt)
217{
218 uint32_t u32Tmp;
219
220 sPt->u32TimeScale = RTC->CLKFMT & RTC_CLKFMT_24HEN_Msk; /* 12/24-hour */
221 sPt->u32DayOfWeek = RTC->WEEKDAY & RTC_WEEKDAY_WEEKDAY_Msk; /* Day of week */
222
223 g_u32hiYear = (RTC->CAL & RTC_CAL_TENYEAR_Msk) >> RTC_CAL_TENYEAR_Pos;
224 g_u32loYear = (RTC->CAL & RTC_CAL_YEAR_Msk) >> RTC_CAL_YEAR_Pos;
225 g_u32hiMonth = (RTC->CAL & RTC_CAL_TENMON_Msk) >> RTC_CAL_TENMON_Pos;
226 g_u32loMonth = (RTC->CAL & RTC_CAL_MON_Msk) >> RTC_CAL_MON_Pos;
227 g_u32hiDay = (RTC->CAL & RTC_CAL_TENDAY_Msk) >> RTC_CAL_TENDAY_Pos;
228 g_u32loDay = (RTC->CAL & RTC_CAL_DAY_Msk);
229
230 g_u32hiHour = (RTC->TIME & RTC_TIME_TENHR_Msk) >> RTC_TIME_TENHR_Pos;
231 g_u32loHour = (RTC->TIME & RTC_TIME_HR_Msk) >> RTC_TIME_HR_Pos;
232 g_u32hiMin = (RTC->TIME & RTC_TIME_TENMIN_Msk) >> RTC_TIME_TENMIN_Pos;
233 g_u32loMin = (RTC->TIME & RTC_TIME_MIN_Msk) >> RTC_TIME_MIN_Pos;
234 g_u32hiSec = (RTC->TIME & RTC_TIME_TENSEC_Msk) >> RTC_TIME_TENSEC_Pos;
235 g_u32loSec = (RTC->TIME & RTC_TIME_SEC_Msk);
236
237 u32Tmp = (g_u32hiYear * 10); /* Compute to 20XX year */
238 u32Tmp += g_u32loYear;
239 sPt->u32Year = u32Tmp + RTC_YEAR2000;
240
241 u32Tmp = (g_u32hiMonth * 10); /* Compute 0~12 month */
242 sPt->u32Month = u32Tmp + g_u32loMonth;
243
244 u32Tmp = (g_u32hiDay * 10); /* Compute 0~31 day */
245 sPt->u32Day = u32Tmp + g_u32loDay;
246
247 if (sPt->u32TimeScale == RTC_CLOCK_12) /* Compute12/24 hour */
248 {
249 u32Tmp = (g_u32hiHour * 10);
250 u32Tmp+= g_u32loHour;
251 sPt->u32Hour = u32Tmp; /* AM: 1~12. PM: 21~32. */
252
253 if (sPt->u32Hour >= 21)
254 {
255 sPt->u32AmPm = RTC_PM;
256 sPt->u32Hour -= 20;
257 }
258 else
259 {
260 sPt->u32AmPm = RTC_AM;
261 }
262
263 u32Tmp = (g_u32hiMin * 10);
264 u32Tmp+= g_u32loMin;
265 sPt->u32Minute = u32Tmp;
266
267 u32Tmp = (g_u32hiSec * 10);
268 u32Tmp+= g_u32loSec;
269 sPt->u32Second = u32Tmp;
270
271 }
272 else
273 {
274 u32Tmp = (g_u32hiHour * 10);
275 u32Tmp += g_u32loHour;
276 sPt->u32Hour = u32Tmp;
277
278 u32Tmp = (g_u32hiMin * 10);
279 u32Tmp += g_u32loMin;
280 sPt->u32Minute = u32Tmp;
281
282 u32Tmp = (g_u32hiSec * 10);
283 u32Tmp += g_u32loSec;
284 sPt->u32Second = u32Tmp;
285 }
