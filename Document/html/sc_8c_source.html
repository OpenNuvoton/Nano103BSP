119 void SC_ResetReader(SC_T *sc)
120 {
121  uint32_t u32Intf;
122 
123  if(sc == SC0)
124  u32Intf = 0;
125  else
126  u32Intf = 1;
127 
128  // Reset FIFO
129  sc->ALTCTL |= (SC_ALTCTL_TXRST_Msk | SC_ALTCTL_RXRST_Msk);
130  // Set Rx trigger level to 1 character, longest card detect debounce period, disable error retry (EMV ATR does not use error retry)
131  sc->CTL &= ~(SC_CTL_RXTRGLV_Msk |
132  SC_CTL_CDDBSEL_Msk |
133  SC_CTL_TXRTY_Msk |
134  SC_CTL_TXRTYEN_Msk |
135  SC_CTL_RXRTY_Msk |
136  SC_CTL_RXRTYEN_Msk);
137  // Enable auto convention, and all three smartcard internal timers
138  sc->CTL |= SC_CTL_AUTOCEN_Msk | SC_CTL_TMRSEL_Msk;
139  // Disable Rx timeout
140  sc->RXTOUT = 0;
141  // 372 clocks per ETU by default
142  sc->ETUCTL = 371;
143  // Enable auto de-activation while card removal
144  sc->PINCTL |= SC_PINCTL_ADACEN_Msk;
145 
146  /* Enable necessary interrupt for smartcard operation */
147  if(u32CardStateIgnore[u32Intf]) // Do not enable card detect interrupt if card present state ignore
148  sc->INTEN = (SC_INTEN_RDAIEN_Msk |
149  SC_INTEN_TERRIEN_Msk |
150  SC_INTEN_TMR0IEN_Msk |
151  SC_INTEN_TMR1IEN_Msk |
152  SC_INTEN_TMR2IEN_Msk |
153  SC_INTEN_BGTIEN_Msk |
154  SC_INTEN_ACERRIEN_Msk);
155  else
156  sc->INTEN = (SC_INTEN_RDAIEN_Msk |
157  SC_INTEN_TERRIEN_Msk |
158  SC_INTEN_TMR0IEN_Msk |
159  SC_INTEN_TMR1IEN_Msk |
160  SC_INTEN_TMR2IEN_Msk |
161  SC_INTEN_BGTIEN_Msk |
162  SC_INTEN_CDIEN_Msk |
163  SC_INTEN_ACERRIEN_Msk);
164 
165  return;
166 }

266 uint32_t SC_GetInterfaceClock(SC_T *sc)
267 {
268  uint32_t reg, freq;
269 
270  if(sc == (SC_T *)SC0)
271  {
272  reg = (CLK->CLKSEL2 & CLK_CLKSEL2_SC0SEL_Msk) >> CLK_CLKSEL2_SC0SEL_Pos;
273  }
274  else
275  {
276  reg = (CLK->CLKSEL2 & CLK_CLKSEL2_SC1SEL_Msk) >> CLK_CLKSEL2_SC1SEL_Pos;
277  }
278 
279  if(reg == (CLK_CLKSEL2_SC0SEL_HXT >> CLK_CLKSEL2_SC0SEL_Pos))
280  {
281  freq = __HXT;
282  }
283  else if(reg == (CLK_CLKSEL2_SC0SEL_PLL >> CLK_CLKSEL2_SC0SEL_Pos))
284  {
285  freq = CLK_GetPLLClockFreq();
286  }
287  else if(reg == (CLK_CLKSEL2_SC0SEL_HIRC >> CLK_CLKSEL2_SC0SEL_Pos))
288  {
289  if(CLK->CLKSEL0 & CLK_CLKSEL0_HIRCSEL_Msk)
290  {
291  freq = __HIRC36M;
292  }
293  else
294  {
295  freq = __HIRC12M;
296  }
297  }
298  else if(reg == (CLK_CLKSEL2_SC0SEL_MIRC >> CLK_CLKSEL2_SC0SEL_Pos))
299  {
300  freq = __MIRC;
301  }
302  else
303  freq = SystemCoreClock;
304 
305  if(sc == (SC_T *)SC0)
306  {
307  freq /= (((CLK->CLKDIV0 & CLK_CLKDIV0_SC0DIV_Msk) >> (CLK_CLKDIV0_SC0DIV_Pos)) + 1);
308  }
309  else
310  {
311  freq /= (((CLK->CLKDIV1 & CLK_CLKDIV1_SC1DIV_Msk) >> (CLK_CLKDIV1_SC1DIV_Pos)) + 1);
312  }
313  return (freq /1000);
314 }
